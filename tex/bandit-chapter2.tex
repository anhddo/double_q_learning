\begin{frame}{$\sigma$-algebra}
    A set $\f \in 2^ \Omega $ is a $\sigma$-algebra if:
    \begin{itemize}
        \item $\Omega  \in  \f $ 
        \item  $A^c  \in \f, \forall A \in \f $ (Closed under complement)
        \item  $\cup_i A_i  \in \f, \forall A_i  \in \f  $  (Close under countable union)
    \end{itemize}

\end{frame}
